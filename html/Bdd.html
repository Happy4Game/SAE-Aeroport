184        query = QSqlQuery(self.db)
185        query.prepare("SELECT nb_passenger, name_ap, name_country FROM (SELECT SUM(seat_nb) AS nb_passenger, name_ap, name_country FROM planeseats AS ps INNER JOIN plane AS p ON p.icao_plane = ps.id_plane INNER JOIN routeavion AS ra ON ra.iata_code = p.iata_plane INNER JOIN routes AS r ON r.id_ra = ra.id_ra INNER JOIN airport ON source_ap = id_ap INNER JOIN country ON country.name_country = airport.country_ap WHERE name_country ILIKE :country GROUP BY name_ap, name_country ORDER BY nb_passenger DESC LIMIT 10 ) AS subquery ORDER BY nb_passenger ASC;")
186        query.bindValue(":country", country)
187
188        if query.exec():
189            nb_passenger_data = [], []

191                nb_passenger = query.value("nb_passenger")
192                airport_name = query.value("name_ap")
193                nb_passenger_data[0].append(nb_passenger)
