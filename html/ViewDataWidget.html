 52    def view_data_bar_nb_passenger_transport(self, country: str) -> None:
 53        """Méthode permettant de créer une data visualisation correspondant au nombre de personne transporté par aeroport
 54
 55        Args:
 56            country (str): Le pays dont on veux connaitre le résultat de la requête
 57        """
 58        #On stock dans une varibale le résultat de la methode
 59        nb_passenger_data = self.bdd.getNbPassengerByAirport(country)
 60
 61        abscisse_pos = np.arange(len(nb_passenger_data[1]) + 1)
 62
 63        # Coche le bouton radio 1
 64        self.__qRadioBbtn1.setChecked(True)
 65
 66        #Transforme les données en str
 67        nb_passenger_str = np.array(nb_passenger_data[0]).astype(str)
 68        nb_passenger_str = np.insert(nb_passenger_str, 0, "0")
 69
 70        # Ajouter le tuple pour la valeur "0"
 71        nb_passenger_data[1].insert(0, " ")
 72
 73        #Paramétrage du graphique
 74        plt.bar(abscisse_pos, nb_passenger_str, align='edge', alpha=0.5, edgecolor='black')
 75        plt.xticks(abscisse_pos, nb_passenger_data[1], rotation=45, ha='right', fontsize=8)
 76        plt.ylabel('Number of Passengers')
 77        plt.title('Passenger Count by Airport in {}'.format(country))
 78
 79        #Ajuster le graphique au layout
 80        plt.tight_layout()
 81
 82        #Nommage et sauvegarde de l'image
 83        temp_file = 'graph_passenger.png'
 84        plt.savefig(temp_file, format='png', dpi=75)
 85        plt.close()
 86
 87        self.image_label : QLabel = QLabel()
 88        self.image_label.setPixmap(QPixmap(temp_file))
 89        self.__layout.addLayout(self.__viewLayoutH)
 90        self.__layout.addWidget(self.image_label, Qt.AlignmentFlag.AlignCenter)

131    def view_data_country(self, country : str) -> None:
132        """Méthode permettant d'afficher une data visualisation de tout les aéroports d'un pays
133
134        Args:
135            country (str): le pays dont on veut connaitre les aéroports
136        """
137        #On stock dans une varibale le résultat de la méthode
138        airport_data = self.bdd.getPositionAeroportOfCountry(country)
139
140        # Création d'un DataFrame Pandas à partir des données des aéroports
141        df = pd.DataFrame(airport_data, columns=["name", "latitude", "longitude"])
142
143        # Création d'une géo-série GeoPandas à partir du DataFrame
144        geometry = [Point(xy) for xy in zip(df['longitude'], df['latitude'])]
145        gdf = gpd.GeoDataFrame(df, geometry=geometry)
146
147        # Chargement de la carte du monde
148        world = gpd.read_file(gpd.datasets.get_path('naturalearth_lowres'))
149
150        # Affichage de la carte du monde et des points des aéroports
151        fig, ax = plt.subplots(figsize=(12, 8))
152        gdf.plot(ax=ax, markersize=10, color='red', alpha=0.7, zorder  = 2)
153        world.boundary.plot(ax=ax, facecolor='lightgray', edgecolor='black', zorder = 1)
154
155        # Paramètres de l'affichage
156        ax.set_title('Airport Locations')
157        ax.set_xlabel('Longitude')
158        ax.set_ylabel('Latitude')
159        ax.set_aspect('equal')
160
161        # Définir les limites de zoom
162        # Obtenir les limites x et y en fonction des coordonnées des aéroports du pays
163        min_x, min_y, max_x, max_y = gdf.total_bounds
164
165        ax.set_xlim(min_x - 10, max_x + 10)  # Ajuster ces valeurs en fonction de la zone d'intérêt
166        ax.set_ylim(min_y - 10, max_y + 10)   # Ajuster ces valeurs en fonction de la zone d'intérêt
167
168
169        # Enregistrer la carte au format PNG
170        temp_file = 'map.png'
171        plt.savefig(temp_file, format='png')
172        plt.close()
173
174        # Charger l'image dans QLabel
175        self.image_label : QLabel = QLabel()
176        self.image_label.setPixmap(QPixmap(temp_file).scaled(500,500))
177        self.__layout.addLayout(self.__viewLayoutH)
178        self.__layout.addWidget(self.image_label, Qt.AlignmentFlag.AlignCenter)
